143 bool jump(Machine *pmach, Instruction instr)
144 {
145  if((instr.instr_generic._regcond != NC && pmach->_cc == CC_U) || instr.instr_generic._regcond > LAST_CONDITION)
146  error(ERR_CONDITION, pmach->_pc - 1);
147  switch(instr.instr_generic._regcond){
148  case NC:
149  return true;
150  case EQ:
151  return pmach->_cc == CC_Z;
152  case NE:
153  return pmach->_cc != CC_Z;
154  case GT:
155  return pmach->_cc == CC_P;
156  case GE:
157  return pmach->_cc == CC_P || pmach->_cc == CC_Z;
158  case LT:
159  return pmach->_cc == CC_N;
160  case LE:
161  return pmach->_cc == CC_N || pmach->_cc == CC_Z;
162  default:
163  error(ERR_CONDITION, pmach->_pc - 1);
164  }
